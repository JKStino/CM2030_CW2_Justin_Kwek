<script>
    function switchToPage(pageNum) {
        document.getElementById('page-main').classList.remove('active-page');
        document.getElementById('page-thermal').classList.remove('active-page');

        if (pageNum === 0) {
            document.getElementById('page-main').classList.add('active-page');
        } else {
            document.getElementById('page-thermal').classList.add('active-page');
        }

        document.querySelectorAll('.page-dot').forEach((dot, i) => {
            dot.classList.toggle('active', i === pageNum);
        });

        if (window.switchPage) window.switchPage(pageNum);
    }
</script>
